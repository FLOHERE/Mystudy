# Serial
- `Stringstr = Serial.readString();`
- Serial : int로 값 반환
- Serial.read : String 으로 값 반환

- 시리얼 모니터를 통해 각도를 입력받고 입력받은 값(0,1,-1)에 따라 switch-case문을 통해 서버모터의 값을 바꾼다
```C
#include<Servo.h>

Servo myServo;

int ServoPin = 3;
int angle = 0;

void setup(){
	Serial.begin(9600);
	myServo.attach(servoPin);
}
void loop(){
	if(Serial.available()){
		int input = Serial.read();
		switch(input){
			case 0:
				angle = 0;
				delay(1000);
			break;
			
			case 1:
				angle+=15;
			break;
			
			case -1:
				angle-=15;
			break;
		}
		myServo.write(angle);
		delay(1000);
	}
}
```
